# 3.1 Write a Python program to show summary statistics of sum, mean,standard deviation, skewness,
#     and kurtosis of any chosen variable.


chosen_data = dataframe['salary_in_usd'] #chosen variable

#to find sum
sum = 0
for i in chosen_data: #iterates each value in chosen_data
    sum = sum + i
print(f"The sum is {sum}.")


#to find mean
count = len(chosen_data)#number of chosen_data
mean = sum / count 
print(f"The mean is {mean}.")


# to find standard deviation
sum_dev = 0
for i in chosen_data:
    sum_dev = sum_dev + (i - mean)**2 # calculates squared deviaton and add to sum_dev
cal = sum_dev / (count - 1)
std_dev = cal ** 0.5
print(f"The standard deviation is {std_dev}")



# to find skewness
skew = 0
for i in chosen_data:
    skew = skew + (i - mean) ** 3 #calculates the skewnsess and add to sum
skew_data = skew / (count * std_dev **3)
print(f"The skewness is {skew_data}.")


# to find kurtosis 
kurtosis = 0
for i in chosen_data:
    dev_from_mean = (i - mean) / std_dev
    kurtosis = kurtosis + dev_from_mean **4
kurtosis = kurtosis / count
print(f"The kurtosis which is calculated manually  is {kurtosis}")
